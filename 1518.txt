class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int drank=numBottles;
        int empty=numBottles;
        while(empty>=numExchange){
            int newbot=empty/numExchange;
            drank+=newbot;
            empty=newbot+(empty%numExchange);
        }
        return drank;
    }
}
