class Solution {
    public boolean canAliceWin(int[] nums) {
       int len=nums.length;
       if(len==1) return true;
       int sum1=0;
       for(int i=0;i<len;i++){
        if(nums[i]<10){
            sum1+=nums[i];
        }else{
            sum1-=nums[i];
        }
       } return sum1!=0;
    }
}
